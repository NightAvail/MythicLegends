#NECROMANCER appear#
character_event = {
	id = adventurer.0
	title = NECROMANCER # Necromancer
	desc = "EVTDESCADVENTURER.0"
	picture = "GFX_evt_plague_doctor"

	trigger = {
		NOT = {
			has_character_flag = adventurer0
		}
		OR = {
			religion = Haelyn
			religion = Sera
			religion = Avani
			religion = Erik
			religion = Nesirie
		}
		OR = {
			tier = DUKE
			tier = KING
			tier = EMPEROR
			tier = COUNT
		}
		any_demesne_province = {
			has_province_modifier = death_plague
		}
	}

	mean_time_to_happen = {
		years = 10
	}

	option = {
		name = adventurer.0.a # Sewers, they say? Better hire an adventurer from these parts...
		random_playable_ruler = {
			limit = {
				any_courtier = {
					culture = ROOT
					trait = adventuring
					NOT = { trait = on_current_quest }
					prisoner = no
				}
			}
			random_courtier = {
				limit = {
					AND = {
						culture = ROOT
						trait = adventuring
						NOT = { trait = on_current_quest }
						prisoner = no
					}
				}
				character_event = {
					id = adventurer.1
				}
			}
		}
	}

	option = {
		name = adventurer.0.b # Sewers, they say? Hire anyone, don't care from where.
		random_playable_ruler = {
			limit = {
				any_courtier = {
					trait = adventuring
					NOT = { trait = on_current_quest }
					prisoner = no
				}
			}
			random_courtier = {
				limit = {
					AND = {
						trait = adventuring
						NOT = {
							trait = on_current_quest
						}
						prisoner = no
					}
				}
				character_event = {
					id = adventurer.1

			}
		}
	}

	option = {
		name = adventurer.0.c # I'm too far away. Have my chancellor hire someone nearby.
		trigger = {
			trait = adventuring
		}
		random_playable_ruler = {
			limit = {
				any_courtier = {
					trait = adventuring
					NOT = { trait = on_current_quest }
					prisoner = no
				}
			}

character_event = {

	title = TROLL_LAIR # Troll Lair
	desc = "EVTDESCADVENTURER.14"
	picture = "GFX_evt_bandits"

	is_triggered_only = yes

	option = {
		name = FORWARD_INTO_THE_DARKNESS # Forward, into the darkness!
		remove_trait = on_current_quest
		wealth = 10
		if = {
		limit = {
					OR = {
						trait = amateur_bard
						trait = amateur_mage
						trait = amateur_priest
						trait = amateur_ranger
						trait = amateur_rogue
						trait = amateur_fighter
						trait = amateur_sorcerer
					}
				}
		change_variable = { which = XP value = 40 }
		}
		if = {
		limit = {
					OR = {
						trait = novice_bard
						trait = novice_mage
						trait = novice_priest
						trait = novice_ranger
						trait = novice_rogue
						trait = novice_fighter
						trait = novice_sorcerer
					}
				}
		change_variable = { which = XP value = 25 }
		}
		if = {
		limit = {
					OR = {
						trait = trained_bard
						trait = trained_mage
						trait = trained_priest
						trait = trained_ranger
						trait = trained_rogue
						trait = trained_fighter
						trait = trained_sorcerer
					}
				}
		change_variable = { which = XP value = 15 }
		}
		# THE START
		#Create the character to fight with
		
		create_random_soldier = {
			name = "Gnoll"
			random_traits = yes
			dynasty = none
			religion = GnollPantheon
			culture = Gnoll
			female = no
			age = 30
		}
		#scope to character we created and send him duel init event
		new_character = {
			set_character_flag = tempchar
			character_event = {id = duelengine.1 days = 1}
			add_trait = novice_fighter
		}
			}

	option = {
		name = adventurer.14.b # No, wait, let's not do this...
		trigger = {
			ai = no
		}
		remove_trait = on_current_quest
		prestige = -50
	}

}
